func lengthOfLongestSubstring(s string) int {
    posMap := make(map[int]int)
    var maxLen int

    for i := 0; i < len(s); i++ {
        charMap := make(map[int]int)
        for s := 0; s <= 127; s++ {
            charMap[s] = 0
        }

        charMap[int(s[i])] = 1
        posMap[i] = 1
        for j := i + 1; j < len(s); j++ {
            v := charMap[int(s[j])]
            if v >= 1 {
                posMap[i] = j - i
                break
            } else {
                charMap[int(s[j])] = 1
                posMap[i] = posMap[i] + 1
            }
        }

        if posMap[i] > maxLen {
            maxLen = posMap[i]
            //maxIndex = i
        }
    }

    return maxLen
}
