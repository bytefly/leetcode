func isPalindrome(x int) bool {
	if x < 0 {
		return false
	}

	var num, prev int
	c := 10
	//convert form lower to higher
	for {
		//get current digit
		ch := x%c - prev
		ch /= c / 10
		//add to the num after multiple with 10
		num = num*10 + ch
		prev = x % c
		//stopped when get to the highest place
		if x/c == 0 {
			break
		}
		c *= 10
	}

	if x != num {
		return false
	}
	return true
}
