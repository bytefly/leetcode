func maxArea(height []int) int {
	maxV := 0
	for i := 0; i < len(height); i++ {
		for j := i + 1; j < len(height); j++ {
			curHeight := height[i]
			if height[i] > height[j] {
				curHeight = height[j]
			}

			curV := (j - i) * curHeight
			if curV > maxV {
				maxV = curV
			}
		}
	}
	return maxV
}
