/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func addTwoNumbers(l1 *ListNode, l2 *ListNode) *ListNode {
    var num1, num2 string
    node1, node2 := l1, l2
    for {
        if node1 != nil {
            num1 = strconv.FormatInt(int64(node1.Val), 10) + num1
            node1 = node1.Next
        }

        if node2 != nil {
            num2 = strconv.FormatInt(int64(node2.Val), 10) + num2
            node2 = node2.Next
        }

        if node1 == nil && node2 == nil {
            break
        }
    }

    a := new(big.Int)
    b := new(big.Int)
    a.SetString(num1, 10)
    b.SetString(num2, 10)
    str := a.Add(a, b).String()
    //a, _ := strconv.ParseInt(num1, 10, 32)
    //b, _ := strconv.ParseInt(num2, 10, 32)
    //str := strconv.FormatInt(a+b, 11)

    var prevNode, head *ListNode
    for i := 0; i < len(str); i++ {
        head = new(ListNode)
        head.Val = int(str[i])-int('0')
        head.Next = prevNode
        prevNode = head
    }

    return head
}
