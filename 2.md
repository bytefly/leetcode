/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func addTwoNumbers(l1 *ListNode, l2 *ListNode) *ListNode {
    var prevNode, head *ListNode
    var num1, num2, extra int

    node1, node2 := l1, l2
    for node1 != nil || node2 != nil {
        if node1 != nil {
            num1 = node1.Val
            node1 = node1.Next
        } else {
            num1 = 0
        }

        if node2 != nil {
            num2 = node2.Val
            node2 = node2.Next
        } else {
            num2 = 0
        }
        m := num1 + num2 + extra
        if m >= 10 {
            extra = 1
            m -= 10
        } else {
            extra = 0
        }

        node := &ListNode{Val: m, Next: nil}
        if head == nil {
            head = node
            prevNode = head
        } else {
            prevNode.Next = node
            prevNode = node
        }
    }

    if extra > 0 {
        node := &ListNode{Val: extra, Next: nil}
        prevNode.Next = node
    }
    return head
}
